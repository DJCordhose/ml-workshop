def calculate_risk(age, speed):
if age < 25:
    if speed > 110:
        return high  # young people, fast cars
    else:
        return medium # young people
        
    if age > 70:
        return high # seniors
    if speed > 145:
        return high # fast cars in general
    # this default vastly simplifies rule set    
return low # otherwise
